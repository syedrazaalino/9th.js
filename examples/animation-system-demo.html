AnimationUtils

KeyframeTrack

AnimationClip

// Create clip const clip = new AnimationClip('move', duration, tracks); // Add track clip.addTrack(track); // Evaluate at time const values = clip.evaluate(time); // Play clip clip.play();

AnimationMixer

Quick Start:
// Basic scene setup with Ninth.js
const scene = new NinthJS.Scene();
const camera = new NinthJS.PerspectiveCamera(75, width/height, 0.1, 1000);
const renderer = new NinthJS.Renderer(canvas);

// Create a rotating cube
const geometry = new NinthJS.BoxGeometry(1, 1, 1);
const material = new NinthJS.MeshStandardMaterial({ color: 0x4488ff });
const cube = new NinthJS.Mesh(geometry, material);
scene.add(cube);

// Animate and render
function animate() {
  requestAnimationFrame(animate);
  cube.rotation.x += 0.01;
  cube.rotation.y += 0.01;
  renderer.render(scene, camera);
}
animate();
